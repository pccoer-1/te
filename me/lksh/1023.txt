### âœ… **Java Code (EmployeeDB.java)**

```java
import java.sql.*;

public class EmployeeDB {
    public static void main(String[] args) {
        try {
            // 1. Load MySQL JDBC Driver
            Class.forName("com.mysql.cj.jdbc.Driver");

            // 2. Connect to Database
            Connection con = DriverManager.getConnection(
                "jdbc:mysql://localhost:3306/yourDB", "root", "password");

            Statement stmt = con.createStatement();

            // 3. Insert Record
            stmt.executeUpdate("INSERT INTO Employee VALUES (104, 'Anusha', 'MH', 28000)");

            // 4. Retrieve Employee Details by SSN
            ResultSet rs = stmt.executeQuery("SELECT * FROM Employee WHERE SSN = 104");
            while (rs.next()) {
                System.out.println("SSN: " + rs.getInt("SSN") +
                                   ", Name: " + rs.getString("Ename") +
                                   ", State: " + rs.getString("state") +
                                   ", Salary: " + rs.getDouble("salary"));
            }

            // 5. Update Employee State
            stmt.executeUpdate("UPDATE Employee SET state='TN' WHERE state='MH'");

            // 6. Delete Employees from 'Gujrat'
            stmt.executeUpdate("DELETE FROM Employee WHERE state='Gujrat'");

            con.close();
            System.out.println("âœ… All operations completed successfully.");
        } catch (Exception e) {
            System.out.println("Error: " + e);
        }
    }
}
```
